\subsection{Cycles}
\tsThe{1.40 (Dirac)}{
    Let $G=(V,E)$ be a graph with $|V|\ge 3$ vertices in which every vertex has at least $|V|/2$ neighbors. Then $G$ is Hamiltonian.
}
\\
\tsThe{1.43 (TSP 2-approximation)}{
    For the metric Traveling Salesman Problem there exists a $2$-approximation algorithm with running time $O(n^2)$.
}
\\
\tsThe{1.42 (TSP $\alpha$-approximation)}{
    If for some $\alpha>1$ there exists an $\alpha$-approximation algorithm for the Traveling Salesman Problem with running time $O(f(n))$, then there also exists an algorithm that decides for all graphs on $n$ vertices whether they are Hamiltonian in time $O(f(n))$.
}
